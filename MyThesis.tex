			\AtPageLowerRight{%
				{\colorbox{gray!30}{%
						\begin{minipage}[b]{\dimexpr\myrightmargin-0.5\marginparsep\relax}
							\vspace*{\paperheight}
							#2
						\end{minipage}%
				}}%
			}%
		}
		{
			\AtPageLowerLeft{%
				{\colorbox{gray!30}{%
						\begin{minipage}[b]{\dimexpr\myrightmargin-0.5\marginparsep\relax}
							\vspace*{\paperheight}
							#1
						\end{minipage}%
				}}%
			}%
		}
	}
